//               MLA 8 FORMATTER CODE


//  Select onOpen in the top bar (near the play button)

//   Press the play button and refresh the document



function InsertHeading() {

  

var body = DocumentApp.getActiveDocument().getBody().setFontFamily(DocumentApp.FontFamily.TIMES_NEW_ROMAN);


var text = body.editAsText();


var date2 = new Date();

var mt = date2.getMonth();

var months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];

var currentM = months[mt]; 

Logger.log("Current Month is: " +currentM); 

var day = date2.getDay();

var year = date2.getYear();

text.insertText(0,day +" " +currentM +" " +year);

text.insertText(0,'English 8\n'); //ADD CLASS AND GRADE HERE (DONT GET RID OF THE \n)

text.insertText(0,'Mrs. Vukorpa\n'); //ADD TEACHER HERE (DON'T GET RID OF THE \n)

text.insertText(0, 'Micah Wagoner\n'); //ADD YOUR NAME HERE (DON'T GET RID OF THE \n)


}



function onOpen(){

  //when the document opens, add a custom menu for auto MLA Formatter

 DocumentApp.getUi().createMenu("Auto MLA Formatter").addItem("Run it!", "InsertHeading").addItem("Notes (PLEASE READ!)", "notes").addItem("Customize to fit your needs!", "howToCustomize").addItem("Report a bug/Send Feedback", "sendFeedback").addToUi()

 

}


function notes(){

}
